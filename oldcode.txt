// navbar.js â€” shared navigation bar for all pages
document.addEventListener("DOMContentLoaded", () => {
  const container = document.getElementById("navbar-container");
  if (!container) return;

  container.innerHTML = `
      <nav id="nav">
        <div class="nav-left">
          <a href="index.html">
            <img src="images/logo.webp" alt="Deals N' Discounts Logo" class="logo" />
          </a>
          <h1>Deals N' Discounts</h1>
        </div>
  
        <div class="nav-center">
          <ul class="nav-links">
            <li><a href="product.html"><i class="fa-solid fa-box"></i> Products</a></li>
            <li><a href="orders.html"><i class="fa-solid fa-truck"></i> Orders</a></li>
            <li><a href="login.html"><i class="fa-solid fa-user"></i> Login</a></li>
          </ul>
        </div>
  
        <div class="nav-right">
          <input type="text" id="searchBar" placeholder="Search products..." />
          <button id="searchBtn"><i class="fa-solid fa-magnifying-glass"></i></button>
        </div>
      </nav>
    `;

  // Highlight active page link
  const path = window.location.pathname.split("/").pop();
  document.querySelectorAll(".nav-links a").forEach((link) => {
    if (link.getAttribute("href") === path) {
      link.classList.add("active");
    }
  });

  // Scroll effect
  window.addEventListener("scroll", () => {
    const nav = document.getElementById("nav");
    if (window.scrollY > 10) nav.classList.add("scrolled");
    else nav.classList.remove("scrolled");
  });

  // Simple search placeholder
  const searchBtn = document.getElementById("searchBtn");
  const searchBar = document.getElementById("searchBar");
  if (searchBtn && searchBar) {
    searchBtn.addEventListener("click", () => {
      const query = searchBar.value.trim();
      if (query) alert(`Searching for: ${query}`);
    });
  }
});
